package com.k4o.arbitrello.data.local

import androidx.room.Dao
import androidx.room.Insert
import androidx.room.OnConflictStrategy
import androidx.room.Query
import kotlinx.coroutines.flow.Flow

/**
 * Interfejs DAO dla operacji na historii wyszukiwania słów.
 */
@Dao
interface WordDao {

    /**
     * Wstawia słowo do historii. W przypadku konfliktu (to samo słowo) zastępuje istniejący wpis
     * w celu aktualizacji znacznika czasu.
     */
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertWord(word: WordEntity)

    /**
     * Zwraca 10 najnowszych wpisów z historii wyszukiwania posortowanych chronologicznie (DESC).
     * Ograniczenie do 10 rekordów służy zachowaniu wydajności i czystości interfejsu.
     */
    @Query("SELECT * FROM search_history ORDER BY timestamp DESC LIMIT 10")
    fun getAllWords(): Flow<List<WordEntity>>

    /**
     * Czyści całą historię wyszukiwania.
     */
    @Query("DELETE FROM search_history")
    suspend fun clearHistory()
}

package com.k4o.arbitrello.data.mapper

import com.k4o.arbitrello.data.local.WordEntity
import com.k4o.arbitrello.model.ArbiterResult

object WordMapper {

    fun toDomain(entity: WordEntity): ArbiterResult {
        return ArbiterResult(
            isValid = entity.isValid,
            checkedWord = entity.checkedWord,
            timestamp = entity.timestamp
        )
    }

    fun toEntity(domain: ArbiterResult): WordEntity {
        return WordEntity(
            checkedWord = domain.checkedWord,
            isValid = domain.isValid,
            timestamp = domain.timestamp
        )
    }
}

package com.k4o.arbitrello.data.repository

import com.k4o.arbitrello.data.local.WordDao
import com.k4o.arbitrello.data.mapper.WordMapper
import com.k4o.arbitrello.remote.PfsService
import com.k4o.arbitrello.model.ArbiterResult
import com.k4o.arbitrello.domain.repository.WordRepository
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.map
import javax.inject.Inject

/**
 * Implementacja repozytorium słów integrująca lokalną bazę danych (WordDao)
 * oraz zewnętrzny serwis PFS (PfsService).
 */
class WordRepositoryImpl @Inject constructor(
    private val wordDao: WordDao,
    private val pfsService: PfsService
) : WordRepository {

    override suspend fun checkWord(word: String): Result<ArbiterResult> {
        return pfsService.checkWords(word)
    }

    override fun getHistory(): Flow<List<ArbiterResult>> {
        return wordDao.getAllWords().map { entities ->
            entities.map { WordMapper.toDomain(it) }
        }
    }

    override suspend fun clearHistory() {
        wordDao.clearHistory()
    }

    override suspend fun saveResult(result: ArbiterResult) {
        val entity = WordMapper.toEntity(result)
        wordDao.insertWord(entity)
    }
}

import javax.inject.Singleton

@Module
@InstallIn(SingletonComponent::class)
object DatabaseModule {

    private const val DATABASE_NAME = "arbitrello_db"

    @Provides
    @Singleton
    fun provideAppDatabase(
        @ApplicationContext context: Context
    ): AppDatabase {
        return Room.databaseBuilder(
            context,
            AppDatabase::class.java,
            DATABASE_NAME
        ).build()
    }

    @Provides
    @Singleton
    fun provideWordDao(database: AppDatabase): WordDao {
        return database.wordDao()
    }
}

     * Pobiera historię sprawdzeń z lokalnej bazy danych.

    /**
     * Czyści całą historię wyszukiwania w bazie danych.
     */

    /**
     * Zapisuje wynik sprawdzenia do historii.
     */
    suspend fun saveResult(result: ArbiterResult)

package com.k4o.arbitrello.domain.usecase

import com.k4o.arbitrello.domain.repository.WordRepository
import com.k4o.arbitrello.model.ArbiterResult
import javax.inject.Inject

/**
 * Use Case odpowiedzialny za weryfikację poprawności słowa.
 * Deleguje zadanie do WordRepository, które zarządza priorytetami źródeł danych.
 */
class CheckWordUseCase @Inject constructor(
    private val repository: WordRepository
) {
    /**
     * Wykonuje sprawdzenie słowa za pośrednictwem repozytorium.
     * Zwraca idiomatynczy obiekt Result zawierający ArbiterResult lub błąd.
     */
    suspend operator fun invoke(word: String): Result<ArbiterResult> {
        return repository.checkWord(word)
    }
}

package com.k4o.arbitrello.remote

import com.k4o.arbitrello.model.AnagramResult
import com.k4o.arbitrello.model.ArbiterResult
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import org.jsoup.Jsoup
import java.util.regex.Pattern

/**
 * Implementacja serwisu PfsService wykorzystująca bibliotekę Jsoup do scrapowania strony PFS.
 */
class JsoupPfsService : PfsService {

    private val mainUrl = "https://www.pfs.org.pl/osps_mobile.php"
    private val apiUrl = "https://www.pfs.org.pl/files/php/osps_funkcje3.php"

    private fun getCsrfToken(): String {
        val mainDoc = Jsoup.connect(mainUrl).get()
        val html = mainDoc.html()
        val pattern = Pattern.compile("(?:sprawdz2|anagramuj2).*?['\"]([a-f0-9]{32})['\"]")
        val matcher = pattern.matcher(html)

        if (!matcher.find()) {
            throw IllegalStateException("Nie znaleziono tokenu CSRF na stronie")
        }

        return matcher.group(1) ?: throw IllegalStateException("Błąd ekstrakcji tokenu")
    }

    override suspend fun checkWords(words: String): Result<ArbiterResult> = withContext(Dispatchers.IO) {
        runCatching {
            val csrfToken = getCsrfToken()

            val responseDoc = Jsoup.connect(apiUrl)
                .data("s", "spr")
                .data("csrfKey", csrfToken)
                .data("slowo_arbiter2", words)
                .ignoreContentType(true)
                .post()

            val resultText = responseDoc.body().text().trim()
            val isValid = resultText == "1"

            ArbiterResult(isValid = isValid, checkedWord = words)
        }
    }

    override suspend fun findAnagrams(letters: String): Result<AnagramResult> = withContext(Dispatchers.IO) {
        runCatching {
            val csrfToken = getCsrfToken()

            val responseDoc = Jsoup.connect(apiUrl)
                .data("s", "ana")
                .data("csrfKey", csrfToken)
                .data("slowo2", letters)
                .ignoreContentType(true)
                .post()

            val responseText = responseDoc.body().text().trim()
            val parts = responseText.split("|").map { it.trim() }.filter { it.isNotEmpty() }

            if (parts.isEmpty()) {
                return@runCatching AnagramResult(emptyList(), "Brak wyników")
            }

            val info = parts[0]
            val wordList = if (parts.size > 1) parts.drop(1) else emptyList()

            AnagramResult(wordList = wordList, info = info)
        }
    }
}

/**
 * AnagramScreen - moduł wyszukiwania anagramów zintegrowany z systemem Pop UI Arbitrello.
 * Obsługuje insets (padding status baru i nawigacji) oraz haptykę.
 */
@Composable
fun AnagramScreen(
    viewModel: AnagramViewModel,
    onBack: () -> Unit
) {
    val uiState by viewModel.uiState.collectAsState()
    var inputLetters by remember { mutableStateOf("") }
    val keyboardController = LocalSoftwareKeyboardController.current
    val haptic = LocalHapticFeedback.current

    // Obsługa zdarzeń haptycznych z ViewModelu
    LaunchedEffect(Unit) {
        viewModel.events.collect { event ->
            when (event) {
                is AnagramEvent.Success -> haptic.performHapticFeedback(HapticFeedbackType.LongPress)
                is AnagramEvent.Error -> {
                    haptic.performHapticFeedback(HapticFeedbackType.TextHandleMove)
                    delay(100)
                    haptic.performHapticFeedback(HapticFeedbackType.TextHandleMove)
                }
            }
        }
    }

    Column(
        modifier = Modifier
            .fillMaxSize()
            .statusBarsPadding()
            .navigationBarsPadding()
            .imePadding()
            .padding(horizontal = 16.dp)
    ) {
        // --- NAGŁÓWEK (Styl Pop) ---
        Box(
            modifier = Modifier
                .fillMaxWidth()
                .height(64.dp),
            contentAlignment = Alignment.Center
        ) {
            IconButton(
                onClick = onBack,
                modifier = Modifier
                    .align(Alignment.CenterStart)
                    .clip(RoundedCornerShape(12.dp))
                    .background(PopMint.copy(alpha = 0.1f))
            ) {
                Icon(
                    imageVector = Icons.Default.ArrowBack,
                    contentDescription = "Powrót",
                    tint = PopMint
                )
            }

            Text(
                text = "ANAGRAMATOR",
                style = MaterialTheme.typography.headlineSmall,
                fontWeight = FontWeight.Black,
                color = PopMint,
                textAlign = TextAlign.Center
            )
        }

        Spacer(modifier = Modifier.height(8.dp))

        // --- INPUT I STEROWANIE ---
        PopTextField(
            value = inputLetters,
            onValueChange = { inputLetters = it.filter { char -> char.isLetter() }.uppercase() },
            label = "TWOJE LITERY",
            placeholder = "NP. ABCDEFG",
            imeAction = ImeAction.Search,
            keyboardActions = KeyboardActions(onSearch = {
                if (inputLetters.isNotBlank()) {
                    viewModel.findAnagrams(inputLetters)
                    keyboardController?.hide()
                }
            }),
            onClear = { inputLetters = "" }
        )

        Spacer(modifier = Modifier.height(16.dp))

        PopButton(
            onClick = {
                viewModel.findAnagrams(inputLetters)
                keyboardController?.hide()
            },
            text = "SZUKAJ SŁÓW",
            color = PopSalmon,
            enabled = inputLetters.isNotBlank() && uiState !is AnagramUiState.Loading,
            modifier = Modifier.fillMaxWidth()
        )

        Spacer(modifier = Modifier.height(24.dp))

        // --- OBSZAR WYNIKÓW ---
        Box(modifier = Modifier.weight(1f)) {
            when (val state = uiState) {
                is AnagramUiState.Loading -> {
                    Column(
                        modifier = Modifier.fillMaxSize(),
                        horizontalAlignment = Alignment.CenterHorizontally,
                        verticalArrangement = Arrangement.Center
                    ) {
                        val composition by rememberLottieComposition(LottieCompositionSpec.RawRes(R.raw.lottie_success))
                        LottieAnimation(
                            composition = composition,
                            iterations = LottieConstants.IterateForever,
                            modifier = Modifier.size(120.dp)
                        )
                        Spacer(modifier = Modifier.height(12.dp))
                        Text(
                            text = "TASUJĘ LITERY...",
                            style = MaterialTheme.typography.labelLarge,
                            fontWeight = FontWeight.Black,
                            color = PopMint,
                            letterSpacing = 2.sp
                        )
                    }
                }
                is AnagramUiState.Success -> {
                    if (state.result.wordList.isEmpty()) {
                        EmptyStateDisplay()
                    } else {
                        Column {
                            Text(
                                text = state.result.info.uppercase(),
                                style = MaterialTheme.typography.labelMedium,
                                fontWeight = FontWeight.Black,
                                color = MaterialTheme.colorScheme.onBackground.copy(alpha = 0.4f),
                                modifier = Modifier.padding(bottom = 12.dp, start = 4.dp)
                            )

                            LazyVerticalGrid(
                                columns = GridCells.Fixed(3),
                                horizontalArrangement = Arrangement.spacedBy(10.dp),
                                verticalArrangement = Arrangement.spacedBy(10.dp),
                                contentPadding = PaddingValues(bottom = 16.dp),
                                modifier = Modifier.fillMaxSize()
                            ) {
                                itemsIndexed(state.result.wordList) { index, word ->
                                    AnagramWordChip(word = word, index = index)
                                }
                            }
                        }
                    }
                }
                is AnagramUiState.Error -> {
                    Column(
                        modifier = Modifier
                            .fillMaxWidth()
                            .padding(24.dp),
                        horizontalAlignment = Alignment.CenterHorizontally
                    ) {
                        Text(
                            text = "UPS! COŚ POSZŁO NIE TAK",
                            color = PopSalmon,
                            fontWeight = FontWeight.Black,
                            textAlign = TextAlign.Center
                        )
                        Text(
                            text = state.message,
                            style = MaterialTheme.typography.bodySmall,
                            color = PopSalmon.copy(alpha = 0.7f),
                            textAlign = TextAlign.Center
                        )
                    }
                }
                else -> {
                    Box(modifier = Modifier.fillMaxSize(), contentAlignment = Alignment.Center) {
                        Text(
                            text = "WRZUĆ LITERY DO MASZYNY!",
                            style = MaterialTheme.typography.titleMedium,
                            fontWeight = FontWeight.Black,
                            color = MaterialTheme.colorScheme.onBackground.copy(alpha = 0.15f),
                            textAlign = TextAlign.Center,
                            letterSpacing = 1.sp
                        )
                    }
                }
            }
        }
    }
}

/**
 * Chip reprezentujący znalezione słowo z animacją bounce i efektem 3D.
 */
@Composable
fun AnagramWordChip(word: String, index: Int) {
    val scale = remember { Animatable(0f) }

    LaunchedEffect(word) {
        delay(index * 30L) // Staggered delay dla efektu fali
        scale.animateTo(
            targetValue = 1f,

package com.k4o.arbitrello.ui.anagram

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.k4o.arbitrello.model.AnagramResult
import com.k4o.arbitrello.remote.PfsService
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.flow.MutableSharedFlow
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.SharedFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asSharedFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.launch
import javax.inject.Inject

sealed interface AnagramUiState {
    object Idle : AnagramUiState
    object Loading : AnagramUiState
    data class Success(val result: AnagramResult) : AnagramUiState
    data class Error(val message: String) : AnagramUiState
}

sealed class AnagramEvent {
    object Success : AnagramEvent()
    data class Error(val message: String) : AnagramEvent()
}

@HiltViewModel
class AnagramViewModel @Inject constructor(
    private val pfsService: PfsService
) : ViewModel() {

    private val _uiState = MutableStateFlow<AnagramUiState>(AnagramUiState.Idle)
    val uiState: StateFlow<AnagramUiState> = _uiState.asStateFlow()

    private val _events = MutableSharedFlow<AnagramEvent>()
    val events: SharedFlow<AnagramEvent> = _events.asSharedFlow()

    fun findAnagrams(letters: String) {
        if (letters.isBlank()) return

        viewModelScope.launch {
            _uiState.value = AnagramUiState.Loading
            pfsService.findAnagrams(letters.uppercase())
                .onSuccess { result ->
                    _uiState.value = AnagramUiState.Success(result)
                    _events.emit(AnagramEvent.Success)
                }
                .onFailure { error ->
                    val msg = error.localizedMessage ?: "Błąd wyszukiwania anagramów"
                    _uiState.value = AnagramUiState.Error(msg)
                    _events.emit(AnagramEvent.Error(msg))
                }
        }
    }

    fun clear() {
        _uiState.value = AnagramUiState.Idle
    }
}

@Composable
fun ResultDisplay(
    word: String,
    isValid: Boolean,
    color: Color,
    progress: Float,
    isErrorState: Boolean = false
) {
    var wordVisible by remember { mutableStateOf(false) }
    var iconVisible by remember { mutableStateOf(false) }
    var statusVisible by remember { mutableStateOf(false) }

    val wordAlpha by animateFloatAsState(
        targetValue = if (wordVisible) 1f else 0f,
        animationSpec = tween(500),
        label = "wordAlpha"
    )
    val iconAlpha by animateFloatAsState(
        targetValue = if (iconVisible) 1f else 0f,
        animationSpec = tween(500),
        label = "iconAlpha"
    )
    val statusAlpha by animateFloatAsState(
        targetValue = if (statusVisible) 1f else 0f,
        animationSpec = tween(500),
        label = "statusAlpha"
    )

    LaunchedEffect(isValid, isErrorState) {
        wordVisible = false
        iconVisible = false
        statusVisible = false
        delay(100)
        wordVisible = true
        delay(300)
        iconVisible = true
        delay(300)
        statusVisible = true
    }

    val composition by rememberLottieComposition(
        LottieCompositionSpec.RawRes(
            if (isErrorState || !isValid) R.raw.lottie_error else R.raw.lottie_success
        )
    )

    val bgFilter = remember(color) { SimpleColorFilter(color.toArgb()) }
    val whiteFilter = remember { SimpleColorFilter(Color.White.toArgb()) }

    val propAllWhite = rememberLottieDynamicProperty(LottieProperty.COLOR_FILTER, whiteFilter, "**")
    val propError8 = rememberLottieDynamicProperty(LottieProperty.COLOR_FILTER, bgFilter, "Shape Layer 8", "**")
    val propError9 = rememberLottieDynamicProperty(LottieProperty.COLOR_FILTER, bgFilter, "Shape Layer 9", "**")
    val propSuccess4 = rememberLottieDynamicProperty(LottieProperty.COLOR_FILTER, bgFilter, "Shape Layer 4", "**")

    val propertiesList = remember(isValid, isErrorState, propAllWhite, propError8, propError9, propSuccess4) {
        mutableListOf<LottieDynamicProperty<*>>().apply {
            add(propAllWhite)
            if (isErrorState || !isValid) {
                add(propError8)
                add(propError9)
            } else {
                add(propSuccess4)
            }
        }
    }

    val dynamicProperties = rememberLottieDynamicProperties(*propertiesList.toTypedArray())

    Column(
        horizontalAlignment = Alignment.CenterHorizontally,
        modifier = Modifier.fillMaxWidth()
    ) {
        Text(
            text = word.uppercase(),
            style = MaterialTheme.typography.displayMedium,
            fontWeight = FontWeight.Black,
            color = Color.White,
            textAlign = TextAlign.Center,
            modifier = Modifier
                .fillMaxWidth()
                .graphicsLayer { alpha = wordAlpha }
        )

        Spacer(modifier = Modifier.height(16.dp))

        Box(
            contentAlignment = Alignment.Center,
            modifier = Modifier
                .size(160.dp)
                .graphicsLayer { alpha = iconAlpha }
        ) {
            CircularProgressIndicator(
                progress = { 1f - progress },
                modifier = Modifier.matchParentSize(),
                color = Color.White,
                trackColor = Color.White.copy(alpha = 0.1f),
                strokeWidth = 8.dp
            )
            if (iconVisible) {
                LottieAnimation(
                    composition = composition,
                    iterations = 1,
                    dynamicProperties = dynamicProperties,
                    modifier = Modifier.size(160.dp)
                )
            }
        }
        Spacer(modifier = Modifier.height(24.dp))
        Text(
            text = when {
                isErrorState -> "BŁĄD POŁĄCZENIA"
                isValid -> "PRAWIDŁOWE"
                else -> "NIEPRAWIDŁOWE"
            },
            style = MaterialTheme.typography.headlineMedium,
            color = Color.White,
            fontWeight = FontWeight.Black,
            textAlign = TextAlign.Center,
            modifier = Modifier
                .fillMaxWidth()
                .graphicsLayer { alpha = statusAlpha }
        )
    }
}

package com.k4o.arbitrello.ui

import android.util.Log
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.k4o.arbitrello.BuildConfig
import com.k4o.arbitrello.domain.repository.WordRepository
import com.k4o.arbitrello.domain.usecase.CheckWordUseCase
import com.k4o.arbitrello.model.ArbiterResult
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.ExperimentalCoroutinesApi
import kotlinx.coroutines.FlowPreview
import kotlinx.coroutines.delay
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
import javax.inject.Inject

private const val TAG = "MainViewModel"

sealed interface ArbiterUiState {
    object Idle : ArbiterUiState
    object Loading : ArbiterUiState
    data class Success(val result: ArbiterResult) : ArbiterUiState
    data class Error(val message: String) : ArbiterUiState
}

@OptIn(FlowPreview::class, ExperimentalCoroutinesApi::class)
@HiltViewModel
class MainViewModel @Inject constructor(
    private val checkWordUseCase: CheckWordUseCase,
    private val wordRepository: WordRepository
) : ViewModel() {

    private val _arbiterUiState = MutableStateFlow<ArbiterUiState>(ArbiterUiState.Idle)
    val arbiterUiState: StateFlow<ArbiterUiState> = _arbiterUiState.asStateFlow()

    // Automatyczna synchronizacja z bazą danych Room (SSOT)
    val history: StateFlow<List<ArbiterResult>> = wordRepository.getHistory()
        .stateIn(
            scope = viewModelScope,
            started = SharingStarted.WhileSubscribed(5000),
            initialValue = emptyList()
        )

    private val _inputFlow = MutableStateFlow("")

    init {
        _inputFlow
            .debounce(1500L)
            .filter { it.isNotBlank() }
            .distinctUntilChanged()
            .flatMapLatest { query ->
                flow {
                    emit(ArbiterUiState.Loading)
                    checkWordUseCase(query)
                        .onSuccess { result ->
                            viewModelScope.launch {
                                delay(3700L)
                                wordRepository.saveResult(result)
                            }
                            emit(ArbiterUiState.Success(result))
                        }
                        .onFailure { exception ->
                            Log.e(TAG, "Error in checkWords for query: $query", exception)
                            val errorMessage = if (BuildConfig.DEBUG) {
                                exception.localizedMessage ?: "Błąd sprawdzania słowa"
                            } else {
                                "Problem z połączeniem lub stroną PFS"
                            }
                            emit(ArbiterUiState.Error(errorMessage))
                        }
                }
            }
            .onEach { state ->
                _arbiterUiState.value = state
            }
            .launchIn(viewModelScope)
    }

    fun onArbiterInputChanged(text: String) {
        val cleaned = text.trim().uppercase()
        _inputFlow.value = cleaned
        if (cleaned.isBlank()) {
            _arbiterUiState.value = ArbiterUiState.Idle
        }
    }

    fun checkWords(words: String) {
        onArbiterInputChanged(words)
    }

    /**
     * Czyści historię wyszukiwania w bazie danych.
     */
    fun clearHistory() {
        viewModelScope.launch {
            wordRepository.clearHistory()
        }
    }
}

/**
 * Pasek nagłówka timera zintegrowany z ArbiterScreen.
 * Zawiera sterowanie czasem, informację o graczu oraz przycisk zamknięcia.
 */
@Composable
fun TimerHeader(viewModel: TimerViewModel) {
    val state by viewModel.uiState.collectAsState()
    val context = LocalContext.current

    // Animacja pulsowania dla stanu końca czasu
    val infiniteTransition = rememberInfiniteTransition(label = "pulse")
    val pulseAlpha by infiniteTransition.animateFloat(
        initialValue = 1f,
        targetValue = 0.5f,
        animationSpec = infiniteRepeatable(
            animation = tween(400, easing = LinearEasing),
            repeatMode = RepeatMode.Reverse
        ),
        label = "alpha_anim"
    )

    // Logika kolorystyki tła
    val baseColor = when {
        state.isEcoMode -> Color.Black
        state.isTimeUp -> PopSalmon
        !state.isRunning -> Color.Gray
        state.timeLeftMs < 10000 -> PopSalmon
        else -> PopMint
    }

    val animatedBgColor by animateColorAsState(
        targetValue = if (state.isTimeUp && !state.isEcoMode) baseColor.copy(alpha = pulseAlpha) else baseColor,
        animationSpec = tween(400),
        label = "color_anim"
    )

    // Reakcja na koniec czasu - wywołanie wibracji (3 sygnały)
    LaunchedEffect(state.isTimeUp) {
        if (state.isTimeUp) {
            viewModel.vibratePhone(context)
        }
    }

    // Formatowanie czasu MM:SS
    val minutes = (state.timeLeftMs / 1000) / 60
    val seconds = (state.timeLeftMs / 1000) % 60
    val timeString = String.format("%02d:%02d", minutes, seconds)

    Row(
        modifier = Modifier
            .statusBarsPadding()
            .fillMaxWidth()
            .heightIn(min = 64.dp)
            .background(animatedBgColor)
            .clickable {
                // Szybka pauza/wznowienie przy kliknięciu w tło
                if (state.isRunning) viewModel.pauseTimer() else viewModel.startTimer()
            }
            .padding(horizontal = 24.dp),
        verticalAlignment = Alignment.CenterVertically,
        horizontalArrangement = Arrangement.SpaceBetween
    ) {
        // Lewa strona: Informacja o graczu i przycisk zmiany
        Row(verticalAlignment = Alignment.CenterVertically) {
            Column {
                Text(
                    text = "GRACZ ${state.activePlayerId} / ${state.totalPlayers}",
                    style = MaterialTheme.typography.labelLarge,
                    fontWeight = FontWeight.Black,
                    color = Color.White
                )
                Text(
                    text = when {
                        state.isTimeUp -> "CZAS MINĄŁ"
                        state.isRunning -> "W TRAKCIE"
                        else -> "PAUZA"
                    },
                    style = MaterialTheme.typography.labelSmall,
                    color = Color.White.copy(alpha = 0.8f)
                )
            }

            Spacer(modifier = Modifier.width(12.dp))

            IconButton(
                onClick = { viewModel.nextPlayer() },
                colors = IconButtonDefaults.iconButtonColors(contentColor = Color.White)
            ) {
                Icon(
                    imageVector = Icons.Default.ArrowForward,
                    contentDescription = "Następny Gracz",
                    modifier = Modifier.size(28.dp)
                )
            }
        }

        // Prawa strona: Stop, Czas i przycisk Start (jeśli pauza)
        Row(
            verticalAlignment = Alignment.CenterVertically,
            horizontalArrangement = Arrangement.End
        ) {
            // Przycisk STOP
            IconButton(
                onClick = { viewModel.stopAndHide() },
                modifier = Modifier.size(48.dp)
            ) {
                Box(
                    modifier = Modifier
                        .size(18.dp)
                        .background(Color.White, RoundedCornerShape(2.dp))
                )
            }

            Spacer(modifier = Modifier.width(8.dp))

            // Tekst czasu z poprawionym wyrównaniem
            Box(
                modifier = Modifier.align(Alignment.CenterVertically),
                contentAlignment = Alignment.Center
            ) {
                Text(
                    text = timeString,
                    style = MaterialTheme.typography.displaySmall.copy(
                        fontSize = 36.sp,
                        fontWeight = FontWeight.Black,
                        platformStyle = PlatformTextStyle(includeFontPadding = false),
                        lineHeightStyle = LineHeightStyle(
                            alignment = LineHeightStyle.Alignment.Center,
                            trim = LineHeightStyle.Trim.Both
                        )
                    ),
                    color = Color.White
                )
            }

            // Przycisk START pojawiający się tylko gdy timer nie działa
            if (!state.isRunning && !state.isTimeUp) {
                Spacer(modifier = Modifier.width(16.dp))
                IconButton(
                    onClick = { viewModel.startTimer() }
                ) {
                    Icon(
                        imageVector = Icons.Default.PlayArrow,
                        contentDescription = "Uruchom Timer",
                        tint = Color.White,
                        modifier = Modifier.size(32.dp)
                    )
                }
            }
        }
    }
}

/**
 * Ekran minutnika z animacjami i obsługą gestów.
 */
@OptIn(ExperimentalFoundationApi::class)
@Composable
fun TimerScreen(viewModel: TimerViewModel) {
    val state by viewModel.uiState.collectAsState()
    val context = LocalContext.current
    val haptic = LocalHapticFeedback.current

    var showTimerSetup by remember { mutableStateOf(false) }

    // Utrzymywanie włączonego ekranu podczas gry
    DisposableEffect(Unit) {
        val window = (context as? Activity)?.window
        window?.addFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)
        onDispose {
            window?.clearFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)
        }
    }

    // Dialog ustawiania sesji (czas i gracze)
    if (showTimerSetup) {
        TimerSetupDialog(
            initialEcoMode = state.isEcoMode,
            onDismiss = { showTimerSetup = false },
            onStart = { minutes, players, isEco ->
                viewModel.setEcoMode(isEco)
                viewModel.setupSession(minutes, players)
                showTimerSetup = false
            }
        )
    }

    // Animacja pulsowania tła po zakończeniu czasu
    val infiniteTransition = rememberInfiniteTransition(label = "timer_effects")
    val flashColor by infiniteTransition.animateColor(
        initialValue = PopSalmon,
        targetValue = Color.White,
        animationSpec = infiniteRepeatable(
            animation = tween(300, easing = LinearEasing),
            repeatMode = RepeatMode.Reverse
        ),
        label = "flash_bg"
    )

    // Animacja efektu "shake" (drgania)
    val shakeOffset by infiniteTransition.animateFloat(
        initialValue = -6f,
        targetValue = 6f,
        animationSpec = infiniteRepeatable(
            animation = tween(40, easing = LinearEasing),
            repeatMode = RepeatMode.Reverse
        ),
        label = "shake_offset"
    )

    // AMOLED: Czysta czerń
    val currentBgColor = when {
        state.isEcoMode -> Color.Black
        state.isTimeUp -> flashColor
        else -> MaterialTheme.colorScheme.background
    }

    Box(
        modifier = Modifier
            .fillMaxSize()
            .background(currentBgColor)
            .padding(24.dp),
        contentAlignment = Alignment.Center
    ) {
        Column(
            horizontalAlignment = Alignment.CenterHorizontally,
            verticalArrangement = Arrangement.SpaceBetween,
            modifier = Modifier.fillMaxHeight().padding(vertical = 40.dp)
        ) {
            // Nagłówek gracza
            Text(
                text = "GRACZ ${state.activePlayerId}",
                style = MaterialTheme.typography.headlineMedium,
                fontWeight = FontWeight.Black,
                color = if (state.isEcoMode) Color.White else if (state.isTimeUp) Color.Black else MaterialTheme.colorScheme.onBackground
            )

            // Centralny licznik z okręgiem (obsługa kliknięcia i przytrzymania)
            Box(
                contentAlignment = Alignment.Center,
                modifier = Modifier
                    .size(300.dp)
                    .graphicsLayer {
                        if (state.isTimeUp) {
                            translationX = shakeOffset
                            translationY = shakeOffset
                        }
                    }
                    .clip(CircleShape)
                    .combinedClickable(
                        onClick = {
                            if (state.isRunning) viewModel.pauseTimer() else viewModel.startTimer()
                        },
                        onLongClick = {
                            haptic.performHapticFeedback(HapticFeedbackType.LongPress)
                            viewModel.nextPlayer()
                        }
                    )
            ) {
                // Wizualizacja postępu
                CircularProgressIndicator(
                    progress = { state.timeLeftMs.toFloat() / state.totalTimeMs.toFloat() },
                    modifier = Modifier.fillMaxSize(),
                    color = when {
                        state.isEcoMode -> Color.White
                        state.timeLeftMs < 10000 -> PopSalmon
                        else -> PopMint
                    },
                    strokeWidth = 16.dp,
                    trackColor = Color.LightGray.copy(alpha = 0.2f)
                )

                // Tekst czasu
                Text(
                    text = formatTime(state.timeLeftMs),
                    style = MaterialTheme.typography.displayLarge.copy(
                        fontSize = 80.sp,
                        fontWeight = FontWeight.Black
                    ),
                    color = when {
                        state.isEcoMode -> Color.White
                        state.timeLeftMs < 10000 -> PopSalmon
                        else -> MaterialTheme.colorScheme.onBackground
                    }
                )
            }

            // Panel sterowania
            Column(
                modifier = Modifier.fillMaxWidth(),
                verticalArrangement = Arrangement.spacedBy(16.dp)
            ) {
                PopButton(
                    onClick = { viewModel.nextPlayer() },
                    text = "NASTĘPNY GRACZ",
                    color = if (state.isEcoMode) Color.DarkGray else PopMint,
                    modifier = Modifier.fillMaxWidth()
                )

                Row(
                    modifier = Modifier.fillMaxWidth(),
                    horizontalArrangement = Arrangement.spacedBy(16.dp)
                ) {
                    PopButton(
                        onClick = { viewModel.resetTimer() },
                        text = "RESET",
                        color = Color.Gray,
                        modifier = Modifier.weight(1f)
                    )

                    PopButton(
                        onClick = { showTimerSetup = true },
                        text = "KONFIGURUJ",
                        color = Color.DarkGray,
                        modifier = Modifier.weight(1f)
                    )
                }
            }
        }
    }
}

/**
 * Formatuje milisekundy na format MM:SS.
 */
private fun formatTime(ms: Long): String {
    val totalSeconds = ms / 1000
    val minutes = totalSeconds / 60
    val seconds = totalSeconds % 60
    return "%02d:%02d".format(minutes, seconds)
}

package com.k4o.arbitrello.ui.timer

import android.content.Context
import android.os.Build
import android.os.VibrationEffect
import android.os.Vibrator
import android.os.VibratorManager
import androidx.lifecycle.SavedStateHandle
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.Job
import kotlinx.coroutines.delay
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.flow.update
import kotlinx.coroutines.launch
import kotlinx.coroutines.isActive
import javax.inject.Inject

data class TimerUiState(
    val timeLeftMs: Long = 60000L,
    val totalTimeMs: Long = 60000L,
    val isRunning: Boolean = false,
    val isTimeUp: Boolean = false,
    val isEcoMode: Boolean = false,
    val activePlayerId: Int = 1,
    val totalPlayers: Int = 2,
    val isTimerVisible: Boolean = false
)

@HiltViewModel
class TimerViewModel @Inject constructor(
    private val savedStateHandle: SavedStateHandle
) : ViewModel() {

    companion object {
        private const val KEY_TIME_LEFT = "TIME_LEFT"
        private const val KEY_ACTIVE_PLAYER = "ACTIVE_PLAYER"
        private const val KEY_IS_RUNNING = "IS_RUNNING"
        private const val KEY_TOTAL_TIME = "TOTAL_TIME"
        private const val KEY_TOTAL_PLAYERS = "TOTAL_PLAYERS"
        private const val KEY_IS_VISIBLE = "IS_VISIBLE"
        private const val KEY_ECO_MODE = "ECO_MODE"
    }

    private val _uiState = MutableStateFlow(
        TimerUiState(
            timeLeftMs = savedStateHandle[KEY_TIME_LEFT] ?: 60000L,
            totalTimeMs = savedStateHandle[KEY_TOTAL_TIME] ?: 60000L,
            activePlayerId = savedStateHandle[KEY_ACTIVE_PLAYER] ?: 1,
            totalPlayers = savedStateHandle[KEY_TOTAL_PLAYERS] ?: 2,
            isRunning = false, // Managed by job
            isTimerVisible = savedStateHandle[KEY_IS_VISIBLE] ?: false,
            isEcoMode = savedStateHandle[KEY_ECO_MODE] ?: false
        )
    )
    val uiState: StateFlow<TimerUiState> = _uiState.asStateFlow()

    private var timerJob: Job? = null

    init {
        // Restore running state after process death
        if (savedStateHandle.get<Boolean>(KEY_IS_RUNNING) == true) {
            startTimer()
        }
    }

    fun setupSession(minutes: Int, players: Int) {
        timerJob?.cancel()
        val totalMs = minutes * 60 * 1000L
        _uiState.update { it.copy(
            totalTimeMs = totalMs,
            timeLeftMs = totalMs,
            totalPlayers = players,
            activePlayerId = 1,
            isRunning = false,
            isTimeUp = false,
            isTimerVisible = true
        ) }
        saveState()
    }

    private fun saveState() {
        val state = _uiState.value
        savedStateHandle[KEY_TIME_LEFT] = state.timeLeftMs
        savedStateHandle[KEY_TOTAL_TIME] = state.totalTimeMs
        savedStateHandle[KEY_ACTIVE_PLAYER] = state.activePlayerId
        savedStateHandle[KEY_TOTAL_PLAYERS] = state.totalPlayers
        savedStateHandle[KEY_IS_RUNNING] = state.isRunning
        savedStateHandle[KEY_IS_VISIBLE] = state.isTimerVisible
        savedStateHandle[KEY_ECO_MODE] = state.isEcoMode
    }

    fun toggleEcoMode() {
        _uiState.update { it.copy(isEcoMode = !it.isEcoMode) }
        saveState()
    }

    fun setEcoMode(enabled: Boolean) {
        _uiState.update { it.copy(isEcoMode = enabled) }
        saveState()
    }

    fun toggleVisibility(visible: Boolean) {
        _uiState.update { it.copy(isTimerVisible = visible) }
        if (!visible) pauseTimer()
        saveState()
    }

    fun stopAndHide() {
        pauseTimer()
        resetTimer()
        toggleVisibility(false)
    }

    fun startTimer() {
        timerJob?.cancel()
        if (_uiState.value.isTimeUp) return

        _uiState.update { it.copy(isRunning = true) }
        saveState()

        timerJob = viewModelScope.launch {
            while (isActive && _uiState.value.timeLeftMs > 0) {
                delay(100L)
                _uiState.update { state ->
                    val newTime = (state.timeLeftMs - 100L).coerceAtLeast(0L)
                    state.copy(timeLeftMs = newTime)
                }
                // Sync important tick state
                savedStateHandle[KEY_TIME_LEFT] = _uiState.value.timeLeftMs
            }
            if (_uiState.value.timeLeftMs <= 0) {
                onTimeUp()
            }
        }
    }

    fun pauseTimer() {
        timerJob?.cancel()
        _uiState.update { it.copy(isRunning = false) }
        saveState()
    }

    fun resetTimer() {
        pauseTimer()
        _uiState.update { it.copy(
            timeLeftMs = it.totalTimeMs,
            isRunning = false,
            isTimeUp = false
        ) }
        saveState()
    }

    fun nextPlayer() {
        timerJob?.cancel()
        _uiState.update { state ->
            val nextId = if (state.activePlayerId >= state.totalPlayers) 1 else state.activePlayerId + 1
            state.copy(
                activePlayerId = nextId,
                timeLeftMs = state.totalTimeMs,
                isTimeUp = false,
                isRunning = false
            )
        }
        saveState()
        startTimer()
    }

    fun vibratePhone(context: Context) {
        val vibrator = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            val vibratorManager = context.getSystemService(Context.VIBRATOR_MANAGER_SERVICE) as VibratorManager
            vibratorManager.defaultVibrator
        } else {
            @Suppress("DEPRECATION")
            context.getSystemService(Context.VIBRATOR_SERVICE) as Vibrator
        }

        val timings = longArrayOf(0, 500, 300, 500, 300, 500)
        val amplitudes = intArrayOf(0, 255, 0, 255, 0, 255)

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            vibrator.vibrate(VibrationEffect.createWaveform(timings, amplitudes, -1))
        } else {
            @Suppress("DEPRECATION")
            vibrator.vibrate(timings, -1)
        }
    }

    private fun onTimeUp() {
        _uiState.update { it.copy(isRunning = false, isTimeUp = true) }
        saveState()
    }

    override fun onCleared() {
        super.onCleared()
        timerJob?.cancel()
    }
}

let debounceTimer;

const mergeFiles = () => {
    console.log('🔄 Wykryto zmiany. Generowanie kontekstu...');
    let fullContent = `--- PROJECT CONTEXT GENERATED AT ${new Date().toLocaleString()} ---\n\n`;

    fullContent += JSON.stringify(CONFIG, null, 2) + '\n\n';

    const walkSync = (dir, filelist = []) => {
        const files = fs.readdirSync(dir);
        files.forEach((file) => {
            const filePath = path.join(dir, file);
            const stat = fs.statSync(filePath);
            
            if (CONFIG.ignorePatterns.some(pattern => filePath.includes(pattern))) return;
